set.seed(1)
G <- 2
N <- 1000
R <- 1000
q <- c(0.6, 0.4)
e <- c(0.3, 0.7)
N_t <- N * q * e
tau_fs <- c(0.2, 0.3)
outcome <-
  foreach (
    g = 1:G, .combine = "rbind"
  ) %do% {
    outcome_g <-
      data.frame(
        g = g,
        y_0 = rnorm(N * q[g], mean = 0, sd = 1),
        y_1 = rnorm(N * q[g], mean = tau_fs[g], sd = 1)
        )
    return(outcome_g)
  }
