  1. Wrapp this into a function. Chekc how to define and use functions.
simulate_and_estimate <-
  function(N, seed) {
    set.seed(seed)
    df <-
      tibble::tibble(
        i = 1:N,
        x = rnorm(N),
        y = x * beta + rnorm(N)
      )
    result <-
      df %>%
      lm(
        data = .,
        formula = y ~ x
      )
    return(
      list(
        df = df,
        result = result
      )
    )
  }

output <- simulate_and_estimate(N = 1000, seed = 1)
